<!DOCTYPE html>

<html>

<head>

  <title>Polyhedral CIF Viewer</title>

  <script src="https://3dmol.org/build/3Dmol-min.js"></script>

  <style>

    body { margin: 0; }

    #viewer {

      width: 100vw;

      height: 100vh;

      position: relative;

    }

  </style>

</head>

<body>

  <div id="viewer"></div>

  <script>

    const element = document.getElementById("viewer");

    const config = { backgroundColor: "white" };

    const viewer = $3Dmol.createViewer(element, config);


    fetch('LNSGYCO STRUCTURE.cif')

      .then(response => response.text())

      .then(cifData => {

        viewer.addModel(cifData, "cif");

        viewer.setStyle({}, { polyhedra: {} });

        viewer.zoomTo();

        viewer.render();

      });

  </script>

</body>

</html>
